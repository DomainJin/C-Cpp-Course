#include <iostream>

int main()
{
    int i,target,size;
    
    printf("Entry size of array:");
    scanf("%d",&size);
    
    int input[size] = {};
    printf("Entry elements: \n");
    
    for(i=0;i<size;i++){
        printf("Input[%d]: ",i);
        scanf("%d",&input[i]);
    }
    
    printf("Entry target: ");
    scanf("%d",&target);
    
    int index = 0;
    for(int k = 0;k<size;k++){
        for(int nd = 1; nd<size-1;nd++){
            if(input[k]+input[k+nd] == target){
                index++;
                printf("Ket qua %d là %d,%d\n",index,k,k+nd);
                // break;
            }
        }
    }
}
